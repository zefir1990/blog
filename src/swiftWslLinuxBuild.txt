Title: Сборка Swift в WSL2 (Linux)
Slug: swift-wsl2-linux-build
Categories: techie,tutorials

<pre><code>sudo apt update && sudo apt upgrade -y</code></pre>

<pre><code>sudo apt install -y \
  git cmake ninja-build clang python3 python3-pip \
  libicu-dev libxml2-dev libcurl4-openssl-dev \
  libedit-dev libsqlite3-dev swig libncurses5-dev \
  pkg-config tzdata rsync</code></pre>

<pre><code>sudo apt install -y llvm lld</code></pre>

<pre><code>git clone https://github.com/apple/swift.git
cd swift

utils/update-checkout --clone</code></pre>

<pre><code>curl -O https://download.swift.org/swiftly/linux/swiftly-$(uname -m).tar.gz && \
tar zxf swiftly-$(uname -m).tar.gz && \
./swiftly init --quiet-shell-followup && \
. "${SWIFTLY_HOME_DIR:-$HOME/.local/share/swiftly}/env.sh" && \
hash -r</code></pre>

<pre><code>utils/build-script \
  --release-debuginfo \
  --swift-darwin-supported-archs="x86_64" \
  --llvm-targets-to-build="X86" \
  --skip-build-benchmarks \
  --skip-test-cmark \
  --skip-test-swift \
  --skip-ios \
  --skip-tvos \
  --skip-watchos \
  --skip-build-libdispatch=false \
  --skip-build-cmark=false \
  --skip-build-foundation \
  --skip-build-lldb \
  --skip-build-xctest \
  --skip-test-swift</code></pre>

<pre><code>export PATH=/root/Sources/3rdparty/build/Ninja-RelWithDebInfoAssert/swift-linux-x86_64/bin/swiftc:$PATH</code></pre>

<pre><code>swift --version</code></pre>

<pre><code>echo "print(\"Hello, World!\")" > hello.swift
swift hello.swift</code></pre>

<pre><code>swiftc hello.swift
./hello</code></pre>
